// The API isBadVersion is defined for you.
// bool isBadVersion(int version);

class Solution {
public:
    int firstBadVersion(int n) {
        int left = 1;
        long long right = n;
        long long  pos=-1;

        while(left <= right){
            long long mid = (left + right)/2;

            if(isBadVersion(mid)) {
                right = mid - 1;
                pos = mid;
            }

            else {
                left = mid + 1;
            }
        }
        return pos;
    }
};
