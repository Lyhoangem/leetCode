class Solution {
public:
    int mySqrt(int x) {
       int left = 1;
       int right = x/2;
       int ans;

       if(x<2) return x;

       while(left <= right){
    long long mid = (left + right)/2;
        if(mid * mid == x){
            return mid;
        }

        else if(mid*mid < x){
            left = mid + 1;
            ans = mid;
        }

        else {
            right = mid -1;
        }
       }
       return ans;
    }
};
